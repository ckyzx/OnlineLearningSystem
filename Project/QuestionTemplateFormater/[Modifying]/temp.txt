// 匹配原始题型行
^[一二三四五六七八九]、
//
[a-zA-Z][^.]
//
^\s+[A-Z]\.
//
^ *[（(].*?$

// 匹配题目行
^[0-9]+\.

// 检查分类行格式
^分类：.*\n
// 检查注释行格式
^\/\/.*\n
// 检查题型行格式
^题型：.*\n

// 检查单选题与多选题的格式
^[0-9]+\..*\( [A-Z]+ \).*\n([A-Z]\..*(\n)*)+
// 检查判断题格式
^[0-9]+\..*\( [×√] \)

// 混合检查单选题、多选题和判断题
(^分类：.*\n)|(^\/\/.*\n)|(^题型：.*\n)|(^[0-9]+\..*\( [×√] \))|(^[0-9]+\..*\( [A-Z]+ \).*\n([A-Z]\..*(\n)*)+)

// 检查案例分析题格式
^([0-9]+\.(.*\n)+?(?=答案：)答案：(\n)?(\s*.*\n)+?)(?=\n)
